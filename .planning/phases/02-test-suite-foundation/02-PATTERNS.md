### `src/util/hash.js` (utility, async transform)

**Analog:** `firebase-init.js` (ESM shape)

```javascript
// src/util/hash.js

/**
 * Trivial hashing for demo purposes (NOT secure — just to avoid plaintext storage).
 * Phase 6 (AUTH-14) replaces with real Firebase Auth — this whole helper goes.
 * @param {string|number|null|undefined} s
 * @returns {Promise<string>}
 */
export async function hashString(s) {
  try {
    const enc = new TextEncoder().encode(String(s));
    const buf = await crypto.subtle.digest("SHA-256", enc);
    return Array.from(new Uint8Array(buf))
      .map((b) => b.toString(16).padStart(2, "0"))
      .join("");
  } catch {
    // fallback
    let h = 0;
    for (const ch of String(s)) h = ((h << 5) - h + ch.charCodeAt(0)) | 0;
    return String(h);
  }
}
```

```javascript
/**
 * @param {object} org
 * @param {string} roundId
 * @param {number} pillarId
 * @param {{ pillars: Array<{ id:number, diagnostics:Array<unknown> }> }} DATA
 * @param {(entry: unknown) => { scale:number }} questionMeta
 */
export function pillarScoreForRound(org, roundId, pillarId, DATA, questionMeta) {
  const p = DATA.pillars.find((pp) => pp.id === pillarId);
  if (!p) return null;
  const byUser = (org.responses || {})[roundId] || {};
  const normalized = [];
  Object.values(byUser).forEach((perPillar) => {
    const perQ = (perPillar || {})[pillarId] || {};
    Object.entries(perQ).forEach(([idx, r]) => {
      if (!Number.isFinite(r.score)) return;
      const meta = questionMeta(p.diagnostics[Number(idx)]);
      if (!meta || !meta.scale) return;
      normalized.push((r.score / meta.scale) * 100);
    });
  });
  if (!normalized.length) return null;
  return Math.round(normalized.reduce((a, b) => a + b, 0) / normalized.length);
}

export function pillarScore(org, pillarId, DATA, questionMeta) {
  return pillarScoreForRound(org, org.currentRoundId, pillarId, DATA, questionMeta);
}

/** @param {object} org */
/** @param {string} roundId */
export function respondentsForRound(org, roundId) {
  const byUser = (org.responses || {})[roundId] || {};
  return Object.keys(byUser);
}

// eslint-disable-next-line no-unused-vars -- Phase 4: remove dead code or wire up call site. See runbooks/phase-4-cleanup-ledger.md
```

```javascript
/**
 * @param {object} org
 * @param {string} roundId
 * @param {string} userId

export function userCompletionPct(org, roundId, userId, DATA) {
  const totalQ = DATA.pillars.reduce((s, p) => s + p.diagnostics.length, 0);
```

```javascript
export function markPillarRead(org, pillarId, user, saveOrg) {
  org.readStates = org.readStates || {};
  org.readStates[user.id] = org.readStates[user.id] || {};
  org.readStates[user.id][pillarId] = iso();
  saveOrg(org);
}

/**
 * @param {{ id:string, role:string, orgId?:string }|null} user
 * @param {{ chatMessages: Array<{ orgId:string, authorId:string, createdAt:any }> }} state
 * @param {(userId:string, orgId:string) => number} lastReadMillis
 * @param {(msg: object) => number} msgMillis
 * @param {(user: object, orgId: string) => number} unreadChatForOrg
 */
export function unreadChatTotal(user, state, lastReadMillis, msgMillis, unreadChatForOrg) {
  if (!user) return 0;
  if (user.role === "client") return unreadChatForOrg(user, user.orgId);
  // internal: count across every org they're seeing
  return (state.chatMessages || []).reduce((n, m) => {
    if (m.authorId === user.id) return n;
    const lastT = lastReadMillis(user.id, m.orgId);
    return msgMillis(m) > lastT ? n + 1 : n;
  }, 0);
}
```

**Note for planner:** `unreadChatTotal` reads `state.chatMessages` and calls `lastReadMillis` / `msgMillis` / `unreadChatForOrg` — all of which stay in the IIFE. The DI list above is intentionally long; that's the cost of byte-identical extraction with state in the IIFE. Phase 4 collapses this when state moves to a `state.js` module.

```javascript
// src/data/cloud-sync.js
// @ts-check
// Phase 2 (D-05): byte-identical extraction from app.js:3556-3593.
// Pitfall 20 / H8 entanglement: cloud wins on overlap (last-writer-wins).
// Phase 5+ subcollection migration rewrites this; TEST-06 pins current behaviour.

/**
 * @param {{
 *   fbReady:            () => boolean,
 *   cloudFetchAllOrgs:  () => Promise<Array<object>|null>,
 *   cloudFetchAllUsers: () => Promise<Array<object>|null>,
 *   cloudPushOrg:       (org: object) => void,
 *   cloudPushUser:      (user: object) => void,
 *   jget:               (k:string, fallback:any) => any,
 *   jset:               (k:string, v:any) => void,
 *   K:                  { orgs:string, users:string, org:(id:string)=>string },
 *   render:             () => void,
 * }} deps
 */
export async function syncFromCloud(deps) {
  const { fbReady, cloudFetchAllOrgs, cloudFetchAllUsers, cloudPushOrg, cloudPushUser, jget, jset, K, render } = deps;
  if (!fbReady()) return;
  const [cloudOrgs, cloudUsers] = await Promise.all([cloudFetchAllOrgs(), cloudFetchAllUsers()]);
  if (cloudOrgs === null || cloudUsers === null) return;

  // ---- Orgs ----
  const localMetas = jget(K.orgs, []);
  const cloudOrgIds = new Set(cloudOrgs.map((o) => o.id));
  localMetas.forEach((meta) => {
    if (!cloudOrgIds.has(meta.id)) {
      const local = jget(K.org(meta.id), null);
      if (local) cloudPushOrg(local);
    }
  });
  cloudOrgs.forEach((o) => {
    if (o && o.id) jset(K.org(o.id), o);
  });
  const newMetas = cloudOrgs.filter((o) => o && o.id).map((o) => ({ id: o.id, name: o.name }));
  localMetas.forEach((m) => {
    if (!cloudOrgIds.has(m.id)) newMetas.push(m);
  });
  jset(K.orgs, newMetas);

  // ---- Users ----
  const localUsers = jget(K.users, []);
  const cloudUserIds = new Set(cloudUsers.map((u) => u.id));
  localUsers.forEach((u) => {
    if (!cloudUserIds.has(u.id)) cloudPushUser(u);
  });
  const merged = [...cloudUsers];
  localUsers.forEach((u) => {
    if (!cloudUserIds.has(u.id)) merged.push(u);
  });
  jset(K.users, merged);

  if (typeof render === "function") render();
}
```

```markdown
## Updating snapshot tests

Snapshot files at `tests/__snapshots__/views/*.html` are committed; the diff in
PR review IS the snapshot governance.

### When a snapshot diff appears

1. `npm test` will fail with a snapshot mismatch.
2. Run `npm test -- -u` locally to update the snapshot files.
3. Inspect the diff — `git diff tests/__snapshots__/`. The diff should match
   the change you intended (e.g., the dashboard header text changed —
   verify the only diff is the header text).
4. Commit the snapshot update in the same PR as the change.
5. Reviewer asserts intentionality of the snapshot diff.

### CI never auto-updates snapshots

The CI invocation is `npm run test:coverage` (no `-u` / `--update`). A snapshot
mismatch in CI is a hard failure. The `-u` flag is a developer tool, never a
CI tool.

## Test runtime budget

Soft target (no enforcement): `npm test` completes in <30s locally and <90s in
CI on a clean clone. If a future PR pushes runtime past 2× the target, the PR
description should justify. Hard timeouts are added in Phase 7 once Cloud
Functions tests + emulator suites push runtime up legitimately.
```

## Shared Patterns

### Pattern A — Module-top header (cross-cutting)

**Source analogs:** `data/pillars.js:1-3` (in-tree style) + `firebase-init.js:1-3` (`@ts-` annotation)

**Apply to:** every new `src/**/*.js` and `tests/**/*.js` file.

```javascript
// <relative path or filename>
// @ts-check
// Phase 2 (D-05): byte-identical extraction from app.js:<lines>. (or: "test for X")
```

The `@ts-check` directive is mandatory per Phase 1 D-29 (`runbooks/phase-4-cleanup-ledger.md` line 21 — Phase 1 carried `@ts-nocheck` only on `app.js` and `firebase-init.js`; new code is checked).

### Pattern B — ESM named-import block (cross-cutting)

**Source analog:** `firebase-init.js:7-35` — the existing in-tree multi-line block-import style.

**Apply to:** every new file that consumes leaf modules; especially `app.js` (the IIFE bridge) and every `tests/**/*.test.js`.

```javascript
import {
  symbolA,
  symbolB,
  symbolC,
} from "./relative/path.js";
```

Always relative paths (`./` or `../`); never bare specifiers (Pitfall 4).

### Pattern C — Eslint-disable comment carry-over (cross-cutting)

**Source analog:** `app.js:33` and `app.js:274` — existing inline disables that point at the cleanup ledger.

**Apply to:** every extracted line range that contains an existing `eslint-disable-next-line`. The disable comment moves verbatim with the function. The cleanup ledger row at `runbooks/phase-4-cleanup-ledger.md:23` (line 33 — `Math.random` in `uid`) and `:31` (line 274 — `no-unused-vars` on `answeredCount`) must be updated to point at the new file path.

### Pattern D — Dependency injection for non-pure leaves (cross-cutting)

**Source analog:** RESEARCH.md § "Pattern 2" lines 543-578.

**Apply to:** `src/domain/scoring.js` (DATA + questionMeta), `src/domain/completion.js` (DATA + pillarScore), `src/domain/unread.js` (saveOrg, commentsFor, state, lastReadMillis, msgMillis, unreadChatForOrg), `src/data/migration.js` (loadUsers, loadOrgMetas, loadOrg, saveOrg, upsertUser, findUser, removeV1ActiveKey), `src/data/cloud-sync.js` (everything), `src/auth/state-machine.js` (loadOrg, findUser, currentSession, INTERNAL_PASSWORD_HASH).

### Pattern E — Caller-side wrapper closure (`app.js` only)

**Source analog:** synthesised from `firebase-init.js:56-80` (`window.FB` namespace shape — same idea: bind dependencies once, expose a stable callable surface).

**Apply to:** every IIFE call site of an extracted function that takes injected deps. Wrapper closures live at the top of the IIFE, immediately after the imports and the `const DATA = window.BASE_LAYERS;` / `const LS = window.localStorage;` block:

```javascript
const pillarScore = (org, pillarId) => _pillarScore(org, pillarId, DATA, questionMeta);
```

Production call sites (`pillarScore(org, p.id)` at every existing usage in `app.js`) stay byte-identical.

### Pattern F — Test file shell (cross-cutting)

**Source analog:** `tests/smoke.test.js` (the entire file — it's the only existing test).

**Apply to:** every `tests/**/*.test.js`.

```javascript
// tests/<dir>/<name>.test.js
// @ts-check
import { describe, it, expect } from "vitest";   // + vi, beforeEach as needed
import { fnUnderTest } from "../../src/<path>.js";

describe("fnUnderTest", () => {
  it("does the thing", () => {
    expect(fnUnderTest(input)).toBe(output);
  });
});
```

**Source analog:** RESEARCH.md lines 1100-1114 (verbatim copy).

**Apply to:** `tests/domain/unread.test.js` (H7) and `tests/data/cloud-sync.test.js` (H8). The header is non-negotiable per CONTEXT.md `<specifics>` second bullet.
