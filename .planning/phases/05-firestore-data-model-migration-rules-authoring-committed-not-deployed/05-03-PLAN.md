/**
 * @typedef {object} ReadState
 * @property {Record<string, *>} pillarReads - per-pillar last-read serverTimestamp values

 * @property {string} [legacyAppUserId]

 * @param {string} orgId
 * @param {string} userId
 * @returns {Promise<ReadState|null>}

export async function getReadState(orgId, userId) {

  return snap.exists() ? /** @type {ReadState} */ (snap.data()) : null;

    doc(db, 'orgs', orgId, 'readStates', userId),

// src/data/messages.js subscribeMessages POST-REWRITE pattern
export function subscribeMessages(orgId, { onChange, onError }) {
  return onSnapshot(
    collection(db, 'orgs', orgId, 'messages'),
    snap => {
      const out = [];
      snap.forEach(d => out.push({ id: d.id, ...d.data() }));
      onChange(out.sort((a, b) => {
        // serverTimestamp values are Firestore Timestamp objects with toMillis()
        const am = a.createdAt?.toMillis ? a.createdAt.toMillis() : 0;
        const bm = b.createdAt?.toMillis ? b.createdAt.toMillis() : 0;
        return am - bm;
      }));
    },
    onError,
  );
}

// Phase 5 Wave 3 (DATA-01 / D-11 / RESEARCH Pattern 3): subcollection access at

import {
  db, collection, doc, getDocs, setDoc, deleteDoc,
  query, where, serverTimestamp,
} from '../firebase/db.js';

/** @returns {string} */
function buildRespId(roundId, userId, pillarId) {
  return `${roundId}__${userId}__${pillarId}`;
}

/**
 * @param {string} orgId
 * @param {string} roundId
 * @param {string} userId
 * @param {number|string} pillarId
 * @returns {Promise<Array<any>>}
 */
export async function listResponses(orgId, roundId, userId, pillarId) {
  const respId = buildRespId(roundId, userId, pillarId);
  const respDoc = await getDocs(query(
    collection(db, 'orgs', orgId, 'responses'),
    where('roundId', '==', roundId),
    where('userId', '==', userId),
    where('pillarId', '==', String(pillarId)),
  ));
  for (const d of respDoc.docs ?? []) {
    if (d.id === respId) return Array.isArray(d.data().values) ? d.data().values : [];
  }
  return [];
}

/**
 * @param {string} orgId
 * @param {string} roundId
 * @param {string} userId
 * @param {number|string} pillarId
 * @param {number} idx
 * @param {*} value
 * @returns {Promise<void>}
 */
export async function saveResponse(orgId, roundId, userId, pillarId, idx, value) {
  const respId = buildRespId(roundId, userId, pillarId);
  const ref = doc(db, 'orgs', orgId, 'responses', respId);
  // Read-modify-write to preserve the values array shape; alternative: store one doc per
  // (round,user,pillar,idx), but the current API contract is "values array per pillar".
  const existingSnap = await import('../firebase/db.js').then(m => m.getDoc(ref));
  const values = existingSnap.exists() && Array.isArray(existingSnap.data().values)
    ? [...existingSnap.data().values]
    : [];
  values[idx] = value;
  await setDoc(ref, {
    roundId, userId, pillarId: String(pillarId),
    values,
    legacyAppUserId: userId,
    updatedAt: serverTimestamp(),
  }, { merge: true });
}

/**
 * @param {string} orgId
 * @param {string} roundId
 * @param {string} userId
 * @param {number|string} pillarId
 * @returns {Promise<void>}
 */
export async function deleteResponse(orgId, roundId, userId, pillarId) {
  const respId = buildRespId(roundId, userId, pillarId);
  await deleteDoc(doc(db, 'orgs', orgId, 'responses', respId));
}
