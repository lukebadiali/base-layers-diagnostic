    import { onCall, HttpsError } from "firebase-functions/v2/https";
    import { logger } from "firebase-functions/logger";
    import { defineSecret } from "firebase-functions/params";
    import { initializeApp, getApps } from "firebase-admin/app";
    import { withSentry } from "../util/sentry.js";
    import { validateInput } from "../util/zod-helpers.js";
    import { ensureIdempotent } from "../util/idempotency.js";
    import { auditEventInput } from "./auditEventSchema.js";
    import { writeAuditEvent } from "./auditLogger.js";

    if (!getApps().length) initializeApp();
    const SENTRY_DSN = defineSecret("SENTRY_DSN");

    export const auditWrite = onCall(
      {
        region: "europe-west2",
        enforceAppCheck: true,                  // FN-07
        serviceAccount: "audit-writer-sa",      // FN-04 (created Wave 1)
        secrets: [SENTRY_DSN],                  // FN-05
        memory: "256MiB",
        timeoutSeconds: 30,
      },
      withSentry(async (request) => {
        if (!request.auth?.uid) {
          throw new HttpsError("unauthenticated", "Authentication required");
        }
        const data = validateInput(auditEventInput, request.data);

        const idempotencyKey = `${request.auth.uid}:${data.type}:${data.target.id}:${data.clientReqId}`;
        await ensureIdempotent(idempotencyKey, "auditWrite", 5 * 60);

        const eventId = crypto.randomUUID();
        const ip = request.rawRequest?.headers["x-forwarded-for"]?.toString().split(",")[0] ?? null;
        const userAgent = request.rawRequest?.headers["user-agent"]?.toString() ?? null;

        await writeAuditEvent(data, {
          eventId,
          actor: {
            uid: request.auth.uid,
            email: request.auth.token.email ?? null,
            role: request.auth.token.role ?? null,
            orgId: request.auth.token.orgId ?? null,
          },
          ip,
          userAgent,
          idempotencyKey,
        });

        logger.info("audit.write", { eventId, type: data.type, actorUid: request.auth.uid });
        return { ok: true, eventId };
      }),
    );
    ```

    Notes:
    - Identity (actor.uid/email/role/orgId) read EXCLUSIVELY from `request.auth.token` — NEVER from `request.data` (Pitfall 17).
    - `ip` derived from `x-forwarded-for` first hop; `userAgent` from raw header.
    - `eventId` server-generated via `crypto.randomUUID()`.
    - `severity` defaults to `"info"` inside `buildAuditEventDoc` if not provided in input.

    import { onDocumentDeleted } from "firebase-functions/v2/firestore";
    import { logger } from "firebase-functions/logger";
    import { initializeApp, getApps } from "firebase-admin/app";
    import { getFirestore } from "firebase-admin/firestore";
    import { writeAuditEvent } from "../auditLogger.js";

    if (!getApps().length) initializeApp();

    export const onOrgDelete = onDocumentDeleted(
      {
        document: "orgs/{orgId}",
        region: "europe-west2",
        serviceAccount: "audit-mirror-sa",       // FN-04
      },
      async (event) => {
        const orgId = event.params.orgId;
        const before = event.data?.data() ?? null;

        // 60s primary-event dedup query (Pattern 4b + Pitfall 7)
        const recent = await getFirestore()
          .collection("auditLog")
          .where("type", "==", "data.org.softDelete")
          .where("target.id", "==", orgId)
          .where("at", ">", new Date(Date.now() - 60_000))
          .limit(1)
          .get();
        if (!recent.empty) {
          logger.info("audit.mirror.skipped", { orgId, reason: "primary-exists" });
          return;
        }

        const eventId = crypto.randomUUID();
        await writeAuditEvent(
          {
            type: "data.org.delete.mirror" as any,    // mirror-only enum value — see schema note below
            severity: "warning",
            target: { type: "org", id: orgId, orgId, snapshot: before },
            clientReqId: eventId,                      // synthesised — no client req
            payload: { source: "trigger", reason: "no-primary-audit-found" },
          },
          {
            eventId,
            actor: { uid: "system", email: null, role: "system", orgId: null },
            ip: null,
            userAgent: null,
            idempotencyKey: `mirror:org:${orgId}:${eventId}`,
          },
        );
        logger.warn("audit.mirror.fired", { orgId });
      },
    );
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser/Client → auditWrite callable | Client payload validated via Zod; actor identity overridden from `request.auth.token` server-side |
| Firestore-trigger event → mirror writer | Server-internal; trust derived from Eventarc; payload `event.data` is the deleted doc snapshot |
| auditLog/{eventId} read API | Restricted to admin role via Firestore Rules `allow read: if isAdmin()` — even audited internal users denied (AUDIT-07) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-02-01 | Tampering | Forged audit event from client | mitigate | (a) Rules `allow write: if false` on auditLog/ (rules-unit-test cell 1-4 pin this); (b) auditWrite overrides actor.{uid,email,role,orgId} from `request.auth.token` (Pitfall 17) |
| T-07-02-02 | Information Disclosure | Audited user reads own log to time next action | mitigate | Rules `allow read: if isAdmin()` only (rules-unit-test cell 7 pins AUDIT-07) |
