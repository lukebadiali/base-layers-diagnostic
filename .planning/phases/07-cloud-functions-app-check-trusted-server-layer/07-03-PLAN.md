    ```js
    // src/firebase/check.js
    // @ts-check
    // Phase 7 (FN-07) body fill — initialises Firebase App Check with reCAPTCHA
    // Enterprise provider per ARCHITECTURE.md §2 + 07-RESEARCH.md Pattern 2.
    //
    // Per-environment site keys: VITE_RECAPTCHA_ENTERPRISE_SITE_KEY pulled from
    // import.meta.env at Vite build time. Debug tokens live only in .env.local
    // (gitignored). Vite tree-shakes the DEV-only branch in production builds
    // (Pitfall 8 mitigation #3).
    //
    // Boot order MUST NOT change — src/firebase/app.js calls initAppCheck(app)
    // immediately after initializeApp() (Phase 4 D-06).

    import { initializeAppCheck, ReCaptchaEnterpriseProvider } from "firebase/app-check";

    const SITE_KEY = import.meta.env.VITE_RECAPTCHA_ENTERPRISE_SITE_KEY ?? "";

    /** @param {import("firebase/app").FirebaseApp} app */
    export function initAppCheck(app) {
      // DEV / emulator path — debug token populated from .env.local
      if (import.meta.env.DEV) {
        const debugToken = import.meta.env.VITE_APPCHECK_DEBUG_TOKEN;
        if (debugToken && typeof self !== "undefined") {
          /** @type {*} */ (self).FIREBASE_APPCHECK_DEBUG_TOKEN = debugToken;
        }
      }

      if (!SITE_KEY) {
        // Fail-closed in PROD if site key missing (would silently bypass App Check
        // without this guard). In DEV, the debug-token path above lets emulator
        // and CI proceed without a real reCAPTCHA Enterprise key.
        if (import.meta.env.PROD) {
          throw new Error("VITE_RECAPTCHA_ENTERPRISE_SITE_KEY not configured (FN-07)");
        }
        return null;
      }

      return initializeAppCheck(app, {
        provider: new ReCaptchaEnterpriseProvider(SITE_KEY),
        isTokenAutoRefreshEnabled: true,
      });
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser/Client → reCAPTCHA Enterprise | Token minted by browser-side `@firebase/app-check`; validated by Google's reCAPTCHA infra |
| App Check token → Firebase backend | Token attached to every Firebase SDK call header; backend per-service config decides enforcement |
| Operator/Developer → debug token | Per-environment debug token in `.env.local` (gitignored) — bypasses reCAPTCHA challenge for local/CI |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
