    ```js
    // src/cloud/retry.js
    // @ts-check
    // Phase 7 (FN-09 / Wave 6) body fill — exponential backoff for callables.
    // Used primarily by src/cloud/audit.js to absorb transient network / 429
    // / unavailable errors without dropping audit events.

    /**
     * @template T
     * @param {() => Promise<T>} fn
     * @param {{ retries?: number, baseMs?: number }} opts
     * @returns {Promise<T>}
     */
    export async function withRetry(fn, opts = {}) {
      const retries = opts.retries ?? 3;
      const baseMs = opts.baseMs ?? 250;
      let lastErr;
      for (let attempt = 0; attempt <= retries; attempt++) {
        try {
          return await fn();
        } catch (err) {
          lastErr = err;
          // Honor non-retryable error codes — do NOT retry on permission-denied,
          // invalid-argument, already-exists (idempotency hit), unauthenticated
          const code = /** @type {any} */ (err)?.code;
          if (code === "permission-denied"
            || code === "invalid-argument"
            || code === "already-exists"
            || code === "unauthenticated") {
            throw err;
          }
          if (attempt === retries) throw err;
          // Exponential backoff with jitter: baseMs * 2^attempt + [0..baseMs] random
          const delay = baseMs * Math.pow(2, attempt) + Math.random() * baseMs;
          await new Promise((resolve) => setTimeout(resolve, delay));
        }
      }
      throw lastErr;
    }
    ```
